// Specification B3 - Input Validation
void moveFunction(char direction, Rooms &currentRoom, Rooms rooms[],
                  int &currentRoomIndex) {
  logFunction("movefunction");

  int newRoomIndex = 0;
  bool validMove = true;

  if (direction == 'N' && currentRoom.exits[0] != ' ') {
    newRoomIndex = currentRoom.exits[0] - '0';
  } else if (direction == 'S' && currentRoom.exits[1] != ' ') {
    newRoomIndex = currentRoom.exits[1] - '0';
  } else if (direction == 'E' && currentRoom.exits[2] != ' ') {
    newRoomIndex = currentRoom.exits[2] - '0';
  } else if (direction == 'W' && currentRoom.exits[3] != ' ') {
    newRoomIndex = currentRoom.exits[3] - '0';
    // Speciﬁcation B2 - Detailed Look
  } else if (direction == 'L') {
    if (currentRoomIndex == 0)
      (cout << "The entrance. It is very spacious; however it is very "
               "ordinary. No lights are on.");
    if (currentRoomIndex == 1)
      (cout << "The kitchen. It is very scary; however it too is very "
               "ordinary. No lights and knives all over...");
    if (currentRoomIndex == 2)
      (cout << "The backyard. Its grassy, and woody; however the swing is "
               "creaking ominously");
    if (currentRoomIndex == 3)
      (cout << "The Living Room. It has no sofas.. The television red and "
               "papers all around. Maybe there is a ghost?");
    if (currentRoomIndex == 4)
      (cout << "The bedroom. It is very small and messy. Who put these books "
               "here? It's disgusting!");

  } else {
    validMove = false;
  }

  // Check if the player wins
  if (currentRoomIndex == 4 && direction == 'E') {
    cout << "You escaped! Winner!" << endl;
    exit(0); // Exit the program
  }

  if (validMove && newRoomIndex < 5) {
    currentRoomIndex = newRoomIndex;
    currentRoom = rooms[currentRoomIndex];
    cout << "You moved to " << currentRoom.room_name << endl << endl;
    spawnmonster(currentRoomIndex);
    roomMove(currentRoom, currentRoomIndex);
    cout << endl << endl;
  } else {
    cout << "Invalid direction!" << endl;
  }
}
